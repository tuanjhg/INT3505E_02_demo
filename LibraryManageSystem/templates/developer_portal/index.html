API Endpoints

Explore our comprehensive API endpoints

GET /api/v2/books List all books with pagination
GET /api/v2/books/{id} Get book details
POST /api/v2/books Create a new book
PUT /api/v2/books/{id} Update book information
DELETE /api/v2/books/{id} Delete a book
GET /api/v2/users List all users
GET /api/v2/users/{id} Get user profile
PUT /api/v2/users/{id} Update user profile
GET /api/v2/borrows List all borrow records
POST /api/v2/borrows Borrow a book
POST /api/v2/borrows/{id}/return Return a borrowed book
POST /api/auth/register Register new user
POST /api/auth/login Login and get JWT token
GET /api/auth/me Get current user info

Code Examples

Get started quickly with our code samples

import requests

# Authentication
response = requests.post("https://api.library.com/api/auth/login", json={
    "username": "demo_user",
    "password": "demo_pass"
})
token = response.json()["token"]

# Get all books
headers = {"Authorization": f"Bearer {token}"}
books = requests.get("https://api.library.com/api/v2/books", headers=headers)

print(books.json())
// Authentication
const loginResponse = await fetch('https://api.library.com/api/auth/login', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        username: 'demo_user',
        password: 'demo_pass'
    })
});
const { token } = await loginResponse.json();

// Get all books
const booksResponse = await fetch('https://api.library.com/api/v2/books', {
    headers: { 'Authorization': `Bearer ${token}` }
});
const books = await booksResponse.json();

console.log(books);
# Login and get token
curl -X POST https://api.library.com/api/auth/login \
     -H "Content-Type: application/json" \
     -d '{"username": "demo_user", "password": "demo_pass"}'

# Get all books
curl -X GET https://api.library.com/api/v2/books \
     -H "Authorization: Bearer YOUR_TOKEN" \
     -H "Content-Type: application/json"

# Create a new book

     -H "Authorization: Bearer YOUR_TOKEN" \
     -H "Content-Type: application/json" \
